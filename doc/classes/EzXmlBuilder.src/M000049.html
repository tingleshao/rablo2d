        def addNode(*args)
                if (args.size == 2)
                        nodeName = args[0]
                        nodeValue = args[1]
                        node = EzXmlNode.new(nodeName, nodeValue, @indent, 1)
                        @rootNodeLst << [ node , @currentIndex ]
                        puts "add root node \"" + nodeName + "\" OK"
                        @currentIndex += 1
                #     updateCont()
                elsif (args.size == 1)
                        node = args[0]
                        node.indent = @indent
                        node.level = 1
                        node.updateChildNodeLevel()
                        @rootNodeLst << [ node, @currentIndex ]
                        puts "add root node \"" + node.name + "\" OK"
                        @currentIndex += 1
                end
        end

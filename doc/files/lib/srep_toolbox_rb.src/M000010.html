def parseSavedSrep(filename)
        srepf = File.open(filename, 'r')
        srepst = srepf.gets.strip
        sreps = []
        matches = srepst.scan(/[0-9]+, [0-9]+, [0-9]+\.[0-9]+/)
        matches.each do |m|
                xyr =  m.scan(/[0-9]+/)
                x = xyr[0].to_f
                y = xyr[1].to_f
                r = xyr[2].to_f + xyr[3].to_f/10
                #~ x = x + Math.sqrt(2) * r
                #~ y = y + Math.sqrt(2) * r
                sreps << [[x, y],[[0,0],[0,0],[0,0]], [r,r,r]]
        end
        return sreps
end

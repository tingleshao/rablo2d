def binaryMatrixToBinary(binmat, size)
        # this is the inverse transform to binaryToBinaryMatrix() function
        bin = ""
        for x in 0..size-1
                for y in 0..size-1
                        if binmat[x][y] == 1
                                bin << "1"
                        else 
                                bin << "0"
                        end
                end
        end
        return bin
end

def binaryToPointList(bin, size)
        # this function converts binary image to point list 
        pointLst = []
        for x in 0..size-1
                for y in 0..size-1
                        index  = x * size + y 
                        if bin[index] == '1'
                                pointLst << [x, y]
                        end
                end
        end
        return pointLst
end

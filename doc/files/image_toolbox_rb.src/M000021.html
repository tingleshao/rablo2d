def pointListToBinary(ptLst, size)
        # this function is the inverse operation of binaryToPointList()
        bin = ''
        copyPtLst = Array.new(ptLst)
        for x in 0..size-1
                for y in 0..size-1
                        if copyPtLst.include?([x,y])
                                bin << "1"
                        else
                                bin << "0"
                        end
                        copyPtLst.delete([x,y])
                end
        end
        return bin
end
